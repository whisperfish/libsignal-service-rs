6//! Handshake trait for key agreement protocols.
7//!
8//! Abstracts over different key agreement protocols (PQXDH, and hypothetical
9//! future variants). The trait separates key agreement from ratchet
10//! initialization and session management.
11//!
12//! See [`crate::pqxdh`] for the current production implementation.
13
14use rand::{CryptoRng, Rng};
15
16use crate::Result;
17
18/// A key agreement protocol used to establish a shared secret during
19/// session initialization.
20///
21/// Implementors handle the cryptographic key agreement (DH computations,
22/// KEM encapsulation/decapsulation, KDF). The resulting session secret is
23/// consumed by the ratchet layer to initialize session state. The initiator
24/// also produces a message that must be sent to the recipient for them to
25/// complete the agreement.
26///
27/// The `initiate` method returns `(InitiatorMessage, SessionSecret)` to
28/// enforce a clean boundary: the message is data for the wire, the secret
29/// is data for the ratchet. These are currently protocol-specific types
30/// but should eventually become opaque byte arrays.
31pub(crate) trait Handshake {
32    /// Parameters for the initiator (constructed from a pre-key bundle).
33    type InitiatorParams;
34
35    /// Parameters for the recipient (constructed from an incoming message).
36    type RecipientParams<'a>;
37
38    /// Data the initiator must send to the recipient for them to complete
39    /// the key agreement (e.g., a KEM ciphertext for PQXDH).
40    type InitiatorMessage;
41
42    /// The shared secret derived from the handshake, consumed by the
43    /// ratchet layer to initialize session state.
44    type SessionSecret;
45
46    /// Perform the initiator side of the key agreement.
47    ///
48    /// Returns the message to send and the session secret to keep.
49    fn initiate<R: Rng + CryptoRng>(
50        params: &Self::InitiatorParams,
51        rng: &mut R,
52    ) -> Result<(Self::InitiatorMessage, Self::SessionSecret)>;
53
54    /// Perform the recipient side of the key agreement.
55    fn accept(params: &Self::RecipientParams<'_>) -> Result<Self::SessionSecret>;
56}
