6//! Double Ratchet implementation.
7//!
8//! The root key, sender chain, and counters are deserialized into typed
9//! fields. Receiver chains are kept in protobuf form and deserialized
10//! lazily on demand.
11//!
12//! References:
13//! - [Double Ratchet spec](https://signal.org/docs/specifications/doubleratchet/)

26/// The state of a Double Ratchet session.
27///
28/// Contains the root key, sending and receiving chains, and cached
29/// message keys for out-of-order messages. Receiver chains are stored
30/// in protobuf form and deserialized lazily on demand.
31#[derive(Clone)]
32pub(crate) struct RatchetState {
33    pub root_key: RootKey,
34    pub sender_chain: Option<SenderChain>,
35    pub receiver_chains: Vec<session_structure::Chain>,
36    pub previous_counter: u32,
37    /// Maximum number of messages we'll skip ahead in a single chain.
38    /// Set to `consts::MAX_FORWARD_JUMPS` for normal sessions,
39    /// `usize::MAX` for self-sessions (note-to-self), following the
40    /// same pattern as SPQR's `max_jump` chain parameter.
41    pub max_forward_jumps: usize,
42}
43
44/// The sending side of a ratchet: our current ephemeral key pair and
45/// the chain key for encrypting outgoing messages.
46#[derive(Clone)]
47pub(crate) struct SenderChain {
48    pub ratchet_key: KeyPair,
49    pub chain_key: ChainKey,
50}
51
52// Receiver chains are stored as raw `session_structure::Chain` protobuf in
53// `RatchetState::receiver_chains`. Only the ratchet key and chain key are
54// deserialized on demand; skipped message keys stay in protobuf form and are
55// only deserialized individually when a matching counter is found.
56
57// ── Serialization bridge ──────────────────────────────────────────────
58
59impl RatchetState {
60    /// Deserialize the ratchet-relevant fields from a `SessionStructure`.
61    ///
62    /// Only reads root key, counters, and sender chain from `session`.
63    /// Receiver chains are passed in separately (moved, not cloned) by
64    /// the caller.
65    ///
66    /// Identity keys, registration IDs, pending pre-key state, and SPQR
67    /// state are owned by higher layers and are not touched here.
68    ///
69    /// `self_session` must be computed by the caller (requires identity
70    /// key comparison, which is not ratchet-layer knowledge).
71    pub(crate) fn from_pb(
72        session: &SessionStructure,
73        self_session: bool,
74        receiver_chains: Vec<session_structure::Chain>,
75    ) -> std::result::Result<Self, InvalidSessionError> {
76        let root_key_bytes: [u8; 32] = session
77            .root_key
78            .as_slice()
79            .try_into()
80            .map_err(|_| InvalidSessionError("invalid root key"))?;
81
82        let sender_chain = session
83            .sender_chain
84            .as_ref()
85            .map(SenderChain::from_pb)
86            .transpose()?;
87
88        Ok(Self {
89            root_key: RootKey::new(root_key_bytes),
90            sender_chain,
91            receiver_chains,
92            previous_counter: session.previous_counter,
93            max_forward_jumps: if self_session {
94                usize::MAX
95            } else {
96                consts::MAX_FORWARD_JUMPS
97            },
98        })
99    }
100
101    /// Write the ratchet state back into a `SessionStructure`.
102    ///
103    /// Only updates the ratchet-relevant fields; all other fields in
104    /// `session` are left unchanged.
105    pub(crate) fn apply_to_pb(self, session: &mut SessionStructure) {
106        let Self {
107            root_key,
108            sender_chain,
109            receiver_chains,
110            previous_counter,
111            max_forward_jumps: _, // not serialized; derived from session context
112        } = self;
113        session.root_key = root_key.key().to_vec();
114        session.previous_counter = previous_counter;
115        session.sender_chain = sender_chain.as_ref().map(SenderChain::to_pb);
116        session.receiver_chains = receiver_chains;
117    }
118}
119
120impl SenderChain {
121    fn from_pb(chain: &session_structure::Chain) -> std::result::Result<Self, InvalidSessionError> {
122        let public_key = PublicKey::deserialize(&chain.sender_ratchet_key)
123            .map_err(|_| InvalidSessionError("invalid sender ratchet public key"))?;
124        let private_key = PrivateKey::deserialize(&chain.sender_ratchet_key_private)
125            .map_err(|_| InvalidSessionError("invalid sender ratchet private key"))?;
126        let chain_key = ChainKey::from_pb(
127            chain
128                .chain_key
129                .as_ref()
130                .ok_or(InvalidSessionError("missing sender chain key"))?,
131        )?;
132        Ok(Self {
133            ratchet_key: KeyPair {
134                public_key,
135                private_key,
136            },
137            chain_key,
138        })
139    }
140
141    fn to_pb(&self) -> session_structure::Chain {
142        session_structure::Chain {
143            sender_ratchet_key: self.ratchet_key.public_key.serialize().to_vec(),
144            sender_ratchet_key_private: self.ratchet_key.private_key.serialize().to_vec(),
145            chain_key: Some(self.chain_key.to_pb()),
146            message_keys: vec![],
147        }
148    }
149}
150
151impl ChainKey {
152    fn from_pb(
153        pb: &session_structure::chain::ChainKey,
154    ) -> std::result::Result<Self, InvalidSessionError> {
155        let key: [u8; 32] = pb
156            .key
157            .as_slice()
158            .try_into()
159            .map_err(|_| InvalidSessionError("invalid chain key"))?;
160        Ok(Self::new(key, pb.index))
161    }
162
163    fn to_pb(&self) -> session_structure::chain::ChainKey {
164        session_structure::chain::ChainKey {
165            index: self.index(),
166            key: self.key().to_vec(),
167        }
168    }
169}

171// ── Operations ───────────────────────────────────────────────────────
172
173impl RatchetState {
174    fn take_root_key(&mut self) -> RootKey {
175        std::mem::replace(&mut self.root_key, RootKey::new([0; 32]))
176    }
177
178    /// Ensure a receiver chain exists for a remote ephemeral key, returning
179    /// its chain key.
180    ///
181    /// If we already have a receiver chain for `their_ephemeral`, return
182    /// its chain key. Otherwise, perform a DH ratchet step: derive a new
183    /// receiver chain from the current root key and the remote ephemeral,
184    /// then generate a fresh sender chain.
185    pub fn ensure_receiver_chain<R: Rng + CryptoRng>(
186        &mut self,
187        their_ephemeral: &PublicKey,
188        csprng: &mut R,
189    ) -> Result<ChainKey> {
190        if let Some(chain_key) = self.find_receiver_chain_key(their_ephemeral)? {
191            return Ok(chain_key);
192        }
193
194        self.dh_ratchet_step(their_ephemeral, csprng)
195    }
196
197    /// Consume the message key for a specific counter value.
198    ///
199    /// If the counter is behind the current chain index, take a cached
200    /// (skipped) key. If it's ahead, advance the chain — caching the
201    /// intermediate keys for out-of-order delivery — up to `max_forward_jumps`.
202    /// Either way, the key is consumed and cannot be retrieved again.
203    pub fn consume_message_key(
204        &mut self,
205        their_ephemeral: &PublicKey,
206        mut chain_key: ChainKey,
207        counter: u32,
208        // The original message type, for error reporting only.
209        original_message_type: CiphertextMessageType,
210        remote_address_for_logging: &str,
211    ) -> Result<MessageKeyGenerator> {
212        let chain_index = chain_key.index();
213
214        if chain_index > counter {
215            // Counter is in the past — look up a cached key.
216            return self
217                .take_skipped_key(their_ephemeral, counter)?
218                .ok_or_else(|| {
219                    log::info!(
220                        "{remote_address_for_logging} Duplicate message for counter: {counter}"
221                    );
222                    SignalProtocolError::DuplicatedMessage(chain_index, counter)
223                });
224        }
225
226        let jump = (counter - chain_index) as usize;
227        if jump > self.max_forward_jumps {
228            log::error!(
229                "{remote_address_for_logging} Exceeded future message limit: {}, index: {chain_index}, counter: {counter}",
230                self.max_forward_jumps,
231            );
232            return Err(SignalProtocolError::InvalidMessage(
233                original_message_type,
234                "message from too far into the future",
235            ));
236        } else if jump > consts::MAX_FORWARD_JUMPS {
237            // This only happens if it is a session with self
238            log::info!(
239                "{remote_address_for_logging} Jumping ahead {jump} messages (index: {chain_index}, counter: {counter})"
240            );
241        }
242
243        // Advance the chain to the target counter, caching skipped keys.
244        while chain_key.index() < counter {
245            self.store_skipped_key(their_ephemeral, chain_key.message_keys());
246            chain_key = chain_key.next_chain_key();
247        }
248
249        // Update the receiver chain to the next key past the one we're returning.
250        self.set_receiver_chain_key(their_ephemeral, chain_key.next_chain_key());
251
252        Ok(chain_key.message_keys())
253    }
254
255    // ── Internals ────────────────────────────────────────────────────
256
257    fn find_receiver_chain_key(
258        &self,
259        their_ephemeral: &PublicKey,
260    ) -> std::result::Result<Option<ChainKey>, InvalidSessionError> {
261        let Some(idx) = self.find_receiver_chain_index(their_ephemeral) else {
262            return Ok(None);
263        };
264        let chain_key_pb = self.receiver_chains[idx]
265            .chain_key
266            .as_ref()
267            .ok_or(InvalidSessionError("missing receiver chain key"))?;
268        Ok(Some(ChainKey::from_pb(chain_key_pb)?))
269    }
270
271    fn dh_ratchet_step<R: Rng + CryptoRng>(
272        &mut self,
273        their_ephemeral: &PublicKey,
274        csprng: &mut R,
275    ) -> Result<ChainKey> {
276        let sender_private_key = self
277            .sender_chain
278            .as_ref()
279            .ok_or(InvalidSessionError("missing sender chain"))?
280            .ratchet_key
281            .private_key;
282
283        // Receiving half-step: root_key + DH(our_sender, their_ephemeral)
284        let current_root_key = self.take_root_key();
285        let (new_root_key, receiver_chain_key) =
286            current_root_key.create_chain(their_ephemeral, &sender_private_key)?;
287
288        // Sending half-step: new_root_key + DH(new_ephemeral, their_ephemeral)
289        let new_sender_key = KeyPair::generate(csprng);
290        let (final_root_key, sender_chain_key) =
291            new_root_key.create_chain(their_ephemeral, &new_sender_key.private_key)?;
292
293        // Record the previous sender chain counter before we replace it.
294        let current_index = self
295            .sender_chain
296            .as_ref()
297            .expect("checked above")
298            .chain_key
299            .index();
300        self.previous_counter = current_index.saturating_sub(1);
301
302        self.root_key = final_root_key;
303
304        self.receiver_chains.push(session_structure::Chain {
305            sender_ratchet_key: their_ephemeral.serialize().to_vec(),
306            sender_ratchet_key_private: vec![],
307            chain_key: Some(receiver_chain_key.to_pb()),
308            message_keys: vec![],
309        });
310        while self.receiver_chains.len() > consts::MAX_RECEIVER_CHAINS {
311            self.receiver_chains.remove(0);
312        }
313
314        self.sender_chain = Some(SenderChain {
315            ratchet_key: new_sender_key,
316            chain_key: sender_chain_key,
317        });
318
319        Ok(receiver_chain_key)
320    }
321
322    fn take_skipped_key(
323        &mut self,
324        their_ephemeral: &PublicKey,
325        counter: u32,
326    ) -> std::result::Result<Option<MessageKeyGenerator>, InvalidSessionError> {
327        let Some(chain_idx) = self.find_receiver_chain_index(their_ephemeral) else {
328            return Ok(None);
329        };
330        let keys = &mut self.receiver_chains[chain_idx].message_keys;
331
332        // Scan by protobuf index field — no deserialization needed for non-matches.
333        let Some(pos) = keys.iter().position(|mk| mk.index == counter) else {
334            return Ok(None);
335        };
336
337        // Remove before deserializing. If from_pb fails the key was corrupt
338        // and unrecoverable — slightly different from main which preserves it,
339        // but the outcome (error) is the same.
340        let key_pb = keys.remove(pos);
341        MessageKeyGenerator::from_pb(key_pb)
342            .map(Some)
343            .map_err(InvalidSessionError)
344    }
345
346    fn store_skipped_key(&mut self, their_ephemeral: &PublicKey, key: MessageKeyGenerator) {
347        let chain_idx = self
348            .find_receiver_chain_index(their_ephemeral)
349            .expect("store_skipped_key called for non-existent chain");
350        let keys = &mut self.receiver_chains[chain_idx].message_keys;
351
352        // TODO: This insert(0) is O(n), making the skip-ahead loop O(n²).
353        // We could switch to push() (appending newest at end) with rposition()
354        // for search and remove(0) for trimming. That makes the common path
355        // O(1) per key. Deferred because it changes serialized key order,
356        // breaking bit-for-bit compatibility with the legacy implementation.
357        keys.insert(0, key.into_pb());
358        if keys.len() > consts::MAX_MESSAGE_KEYS {
359            keys.pop();
360        }
361    }
362
363    fn set_receiver_chain_key(&mut self, their_ephemeral: &PublicKey, chain_key: ChainKey) {
364        let chain_idx = self
365            .find_receiver_chain_index(their_ephemeral)
366            .expect("set_receiver_chain_key called for non-existent chain");
367        self.receiver_chains[chain_idx].chain_key = Some(chain_key.to_pb());
368    }
369
370    fn find_receiver_chain_index(&self, their_ephemeral: &PublicKey) -> Option<usize> {
371        self.receiver_chains.iter().position(|chain| {
372            match PublicKey::deserialize(&chain.sender_ratchet_key) {
373                Ok(key) => &key == their_ephemeral,
374                Err(_) => {
375                    log::warn!("skipping corrupt receiver chain with invalid ratchet key");
376                    false
377                }
378            }
379        })
380    }
