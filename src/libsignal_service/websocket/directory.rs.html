1//! Contact Discovery Service (CDSI) authentication
2//!
3//! Provides authentication credentials for CDSI contact lookup operations.
4
5use crate::content::ServiceError;
6use crate::websocket::{Identified, SignalWebSocket};
7use reqwest::Method;
8use serde::Deserialize;
9
10/// CDSI authentication credentials
11#[derive(Debug, Deserialize)]
12pub struct CdsiAuth {
13    pub username: String,
14    pub password: String,
15}
16
17impl SignalWebSocket<Identified> {
18    /// Get CDSI authentication credentials from the chat server.
19    ///
20    /// Returns username/password credentials for establishing an
21    /// authenticated connection to the Contact Discovery Service.
22    ///
23    /// # Returns
24    /// * `Ok(CdsiAuth)` - Authentication credentials
25    /// * `Err(ServiceError)` - Network or protocol error
26    ///
27    /// # Example
28    /// ```no_run
29    /// # use libsignal_service::websocket::{SignalWebSocket, Identified};
30    /// # async fn example(mut ws: SignalWebSocket<Identified>) {
31    /// let auth = ws.get_cdsi_auth().await.unwrap();
32    /// // Use auth.username and auth.password for CDSI connection
33    /// # }
34    /// ```
35    pub async fn get_cdsi_auth(&mut self) -> Result<CdsiAuth, ServiceError> {
36        let response = self
37            .http_request(Method::GET, "/v2/directory/auth")?
38            .send()
39            .await?
40            .service_error_for_status()
41            .await?;
42
43        response.json().await
44    }
45}
